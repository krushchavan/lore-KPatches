Patch summary
This patch series adds support for CB_NOTIFY callbacks in directory delegations, allowing clients to register interest in a directory and receive notifications about changes within it without losing its lease. The patches include bugfixes, new tracepoints, and the addition of a new FSNOTIFY_EVENT_RENAME data type to handle cross-directory rename events. The series builds on previous work on directory delegation and has been tested with pynfs, but further client-side implementation is needed for complete testing.

+/**
+ * inode_lease_ignore_mask - return union of all ignored inode events for this inode
+ * @inode: inode of which to get ignore mask
+ *
+ * Walk the list of leases, and return the result of all of
+ * their FL_IGN_DIR_* bits or'ed together.
+ */
+u32
+inode_lease_ignore_mask(struct inode *inode)
+{
+	struct file_lock_context *ctx;
+	struct file_lock_core *flc;
+	u32 mask = 0;
+
+	ctx = locks_inode_context(inode);
+	if (!ctx)
+		return 0;
+
+	spin_lock(&ctx->flc_lock);
+	list_for_each_entry(flc, &ctx->flc_lease, flc_list) {
+		mask |= flc->flc_flags & IGNORE_MASK;
+		/* If we already have everything, we can stop */
+		if (mask == IGNORE_MASK)

+	}

+	return mask;
+}
